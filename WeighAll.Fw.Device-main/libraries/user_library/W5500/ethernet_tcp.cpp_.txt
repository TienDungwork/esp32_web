void Ethernet_Reset(const uint8_t resetPin) {
    pinMode(resetPin, OUTPUT);
    digitalWrite(resetPin, HIGH);
    delay(250);
    digitalWrite(resetPin, LOW);
    delay(50);
    digitalWrite(resetPin, HIGH);
    delay(350);
}

bool Ethernet_Initialization_Assertion(String *assertion) {
    switch (Ethernet.hardwareStatus()) {
    case EthernetNoHardware:
        PrintDebugLn("[W5500] Ethernet No Hardware");
        *assertion = "[W5500] Ethernet No Hardware";
        return false;
    case EthernetW5100:
        PrintDebugLn("[W5500] Ethernet W5100 Discovery !");
        break;
    case EthernetW5200:
        PrintDebugLn("[W5500] Ethernet W5200 Discovery !");
        break;
    case EthernetW5500:
        PrintDebugLn("[W5500] Ethernet W5500 Discovery !");
        break;
    }

    switch (Ethernet.linkStatus()) {
    case Unknown:
        PrintDebug("[W5500] Link status: ");
        PrintDebugLn("[W5500] Unknown");
        PrintDebugLn("[W5500] Hardware error !");
        *assertion = "[W5500] Hardware error";
        return false;
    case LinkON:
        PrintDebug("[W5500] Link status: ");
        PrintDebugLn("[W5500] ON");
        break;
    case LinkOFF:
        PrintDebug("[W5500] Link status: ");
        PrintDebugLn("[W5500] OFF");
        PrintDebugLn("[W5500] The network cable is not connected !");
        *assertion = "[W5500] Please insert the network cable";
        return false;
    }

    PrintDebugLn("[W5500] Trying to get an IP address using DHCP...");
    if (Ethernet.begin(mac, 10000, 10000) == 0) {
        *assertion = "DHCP configuration failed";
        return false;
    } else {
        PrintDebugLn("-------------------------");
        PrintDebugLn("[INFO] Configuring random DHCP successfully !");
        PrintDebugLn("");
        PrintDebug("[DHCP] IP Address: ");
        PrintDebugLn(Ethernet.localIP());
        PrintDebug("[DHCP] Subnet Mask: ");
        PrintDebugLn(Ethernet.subnetMask());
        PrintDebug("[DHCP] Gateway: ");
        PrintDebugLn(Ethernet.gatewayIP());
        PrintDebug("[DHCP] DNS: ");
        PrintDebugLn(Ethernet.dnsServerIP());
        PrintDebugLn("-------------------------");
        PrintDebugLn("");
    }
    return true;
}

int udp_receive(uint8_t *buf, uint16_t buf_size, IPAddress *src_ip, uint16_t *src_port) {
    int packetSize = udp.parsePacket();
    if (packetSize > 0) {
        *src_ip = udp.remoteIP();
        *src_port = udp.remotePort();
        int len = udp.read(buf, buf_size - 1); // Chừa chỗ cho '\0'
        if (len > 0) {
            buf[len] = '\0'; // Chuỗi kết thúc
            return len;
        }
    }
    return 0;
}

void udp_handle_packet(uint8_t *buf, int len, IPAddress src_ip, uint16_t src_port) {
    PrintDebugf("[UDP] Received %d bytes from %s:%d\n", len, src_ip.toString().c_str(), src_port);
    PrintDebugf("[UDP] Data: %s\n", buf);

    // Khởi tạo parser JSMN
    jsmn_parser parser;
    jsmn_init(&parser);
    
    // Tạo mảng token (giả sử tối đa 10 token cho JSON đơn giản)
    #define MAX_TOKENS 10
    jsmntok_t tokens[MAX_TOKENS];
    
    // Phân tích JSON
    int result = jsmn_parse(&parser, (char*)buf, len, tokens, MAX_TOKENS);
    if (result < 0) {
        PrintDebugf("[UDP] JSON parse error: %d\n", result);
        return;
    }

    // Kiểm tra nếu JSON là một object
    if (result < 1 || tokens[0].type != JSMN_OBJECT) {
        PrintDebugLn("[UDP] JSON is not a valid object");
        return;
    }

    // Biến để lưu trữ ip và port
    char ip_str[16] = {0}; // Chuỗi IP (tối đa 15 ký tự, ví dụ: "192.168.1.123")
    uint64_t port = 0;

    // Duyệt qua các token để tìm "ip" và "port"
    int code = 0;  // mặc định

    for (int i = 1; i < result; i += 2) {
        if (jsoneq((char*)buf, &tokens[i], "code") == 0) {
            json_parse_uint((char*)buf, &tokens[i + 1], &code);
        } else if (jsoneq((char*)buf, &tokens[i], "IP") == 0) {
            json_parse_string((char*)buf, &tokens[i + 1], ip_str);
        } else if (jsoneq((char*)buf, &tokens[i], "Port") == 0) {
            json_parse_uint64((char*)buf, &tokens[i + 1], &port);
        }
    }


    // Kiểm tra và cập nhật serverIP và serverPort
    if (strlen(ip_str) > 0 && port > 0 && port <= 65535) {
        serverIP.fromString(ip_str); // Cập nhật serverIP
        serverPort = (int)port;      // Cập nhật serverPort
        PrintDebugf("[UDP] Updated TCP server config: IP=%s, Port=%d\n", 
                      serverIP.toString().c_str(), serverPort);

        save_server_config(ip_str, serverPort);

        // Ngắt kết nối TCP hiện tại (nếu có) và kết nối lại với cấu hình mới
        tcpClient.stop();
        delay(1000);
        if (tcpClient.connect(serverIP, serverPort)) {
            PrintDebugLn("[TCP] Reconnected with new server config");
            EventQueue_Push(EVENT_TCP_CONNECTED, NULL); // Đẩy sự kiện kết nối TCP
        } else {
            PrintDebugLn("[TCP] Failed to reconnect with new server config");
            EventQueue_Push(EVENT_TCP_DISCONNECTED, NULL);
        }
    } else {
        PrintDebugLn("[UDP] Invalid IP or port in JSON");
    }

    udp_send_response(src_ip, src_port, "[UDP]: Received your broadcast\r\n");
}

void udp_process(void) {
    uint8_t udp_buf[UDP_BUFFER_SIZE];
    IPAddress src_ip;
    uint16_t src_port;
    int len = udp_receive(udp_buf, UDP_BUFFER_SIZE, &src_ip, &src_port);
    if (len > 0) {
        udp_handle_packet(udp_buf, len, src_ip, src_port);
    }
}

void udp_send_response(IPAddress dst_ip, uint16_t dst_port, const char *msg) {
    udp.beginPacket(dst_ip, dst_port);
    udp.write((const uint8_t *)msg, strlen(msg));
    udp.endPacket();
}

void save_server_config(const char* ip_str, uint16_t port) {
    preferences.begin("server_cfg", false); // "false" = RW mode
    preferences.putString("ip", ip_str);
    preferences.putUShort("port", port);
    preferences.end();
    PrintDebugf("[FLASH] Saved IP=%s, Port=%d\n", ip_str, port);
}

bool load_server_config(IPAddress &ip, uint16_t &port) {
    preferences.begin("server_cfg", true); // "true" = read-only
    String ip_str = preferences.getString("ip", "");
    port = preferences.getUShort("port", 0);
    preferences.end();

    if (ip_str.length() > 0 && port > 0 && port <= 65535) {
        ip.fromString(ip_str);
        PrintDebugf("[FLASH] Loaded IP=%s, Port=%d\n", ip_str.c_str(), port);
        return true;
    } else {
        PrintDebugLn("[FLASH] No valid server config found.");
        return false;
    }
}

bool EventQueue_Push(AppEvent type, const char* data) {
    uint8_t next = (event_tail + 1) % EVENT_QUEUE_SIZE;
    if (next == event_head) return false; // full
    event_queue[event_tail].type = type;
    if (data)
        strncpy(event_queue[event_tail].data, data, RECEIVE_BUF_SIZE - 1);
    else
        event_queue[event_tail].data[0] = '\0';
    event_tail = next;
    return true;
}

bool EventQueue_PushOverwrite(AppEvent type, const char* data) {
    uint8_t next = (event_tail + 1) % EVENT_QUEUE_SIZE;
    if (next == event_head) {
        event_tail = (event_tail - 1 + EVENT_QUEUE_SIZE) % EVENT_QUEUE_SIZE;
    }
    event_queue[event_tail].type = type;
    if (data)
        strncpy(event_queue[event_tail].data, data, RECEIVE_BUF_SIZE - 1);
    else
        event_queue[event_tail].data[0] = '\0';
    event_tail = (event_tail + 1) % EVENT_QUEUE_SIZE;
    return true;
}

bool EventQueue_Pop(Event* e) {
    if (event_head == event_tail) return false; // empty
    *e = event_queue[event_head];
    event_head = (event_head + 1) % EVENT_QUEUE_SIZE;
    return true;
}

/******************************************************************************/

void TCP_Client_HandlePing() {
    if (!waitingForPong && (millis() - lastDataTxRxTime >= 5000)) {
        PrintDebugLn("[\r\nEVEN] Send ping >= 5000ms");
        EventQueue_Push(EVENT_TCP_SEND_PING, NULL);
        lastPingSentTime = millis();
        waitingForPong = true;
    }
    if (waitingForPong && (millis() - lastPingSentTime  > 3000)) {
        PrintDebugLn("[EVENT] Ping timeout (3000ms) -> reconnecting");
        EventQueue_Push(EVENT_TCP_DISCONNECTED, NULL);
        waitingForPong = false;
    }
}

void TCP_Client_HandleReceive() {
    if (!tcpClient.connected()) return;
    if (tcpClient.available()) {
        String message = "";
        while (tcpClient.available()) {
            char c = tcpClient.read();
            message += c;
        }
        message.trim();
        if (message.length() > 0) {
            EventQueue_Push(EVENT_TCP_DATA_RECEIVED, message.c_str());
        }
    }
}

void TCP_Client_CheckConnection() {
    if (!tcpClient.connected()) {
        EventQueue_Push(EVENT_TCP_DISCONNECTED, NULL);
    } else {
        EventQueue_Push(EVENT_TCP_CONNECTED, NULL);
    }
}

void GFX_Print_Ethernet_Info_Loop(void) {

//#if DEVICE_HAS_DISPLAY
    if (millis() > CycleTime) {
        if (Ethernet.linkStatus() != LinkON) {
            Ethernet_Relay_OP.initialization.flag = false;
        }
        GFX_Update_Ethernet_Status();
        CycleTime = millis() + 500;
    }

    if (millis() > CycleTime_2) {
        if (Ethernet.linkStatus() != LinkON) {
            Ethernet_Relay_OP.initialization.flag = false;
            PrintDebugLn("[Ethernet] Link lost, reinitializing...");
        }
        CycleTime_2 = millis() + 3000;
    }
//#endif  // DEVICE_HAS_DISPLAY

    //udp_process(); // Xử lý các gói tin UDP broadcast
    //TCP_Client_CheckConnection();
    //TCP_Client_HandleReceive();
    //TCP_Client_HandlePing();

    soft_timer_update();

    Event e;
    while (EventQueue_Pop(&e)) {
        HandleEvent(&e);
    }
}

void sendHttpRequest(String path) {
    if (httpClient.connect(LED_SERVER_IP, LED_SERVER_PORT)) {
        httpClient.print(String("GET ") + path + " HTTP/1.1\r\n" +
                         "Host: " + LED_SERVER_IP + "\r\n" +
                         "Connection: close\r\n\r\n");
        PrintDebugLn("\r\n[HTTP] Sent to LED: " + path);
    } else {
        PrintDebugLn("\r\n[HTTP] Failed to connect to LED");
    }
    delay(200);
    httpClient.stop();
}
