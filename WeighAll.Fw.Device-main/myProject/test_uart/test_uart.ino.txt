#include <Arduino.h>
#include "pin_config.h"

#define SAMPLE_COUNT 10

HardwareSerial SerialPort1(1); // UART2, giả sử Serial2 là SerialPort1
char DATA[256] = {0};
String rs232_input = "";
String buffer = ""; // Buffer cho dữ liệu RS232
float samples[SAMPLE_COUNT];
int sampleIndex = 0;
bool collecting = false;
uint32_t temp = 0;
uint32_t CycleTime = 0;

// Hàm tìm giá trị xuất hiện nhiều nhất
float findMostFrequent(float arr[], int n) {
    int maxCount = 0;
    float mostFrequent = arr[0];

    for (int i = 0; i < n; i++) {
        int count = 1;
        for (int j = i + 1; j < n; j++) {
            if (fabs(arr[i] - arr[j]) < 0.0001f) {
                count++;
            }
        }

        if (count > maxCount) {
            maxCount = count;
            mostFrequent = arr[i];
        }
    }

    return mostFrequent;
}

// Hàm giả lập đẩy sự kiện vào EventQueue
void EventQueue_Push(int eventType, const char* data) {
    // Giả lập in ra Serial để kiểm tra
    Serial.print("Event Pushed: Type=");
    Serial.print(eventType);
    Serial.print(", Data=");
    Serial.println(data);
}

// Hàm xử lý dữ liệu RS232 (kết hợp cả hai logic)
void receiveRS232Data() {
    while (SerialPort1.available()) {
        char c = SerialPort1.read();
        buffer += c;

        if (c == '=') {
            if (buffer.length() > 1) { // Kiểm tra buffer có dữ liệu trước dấu '='
                String valueStr = buffer.substring(0, buffer.length() - 1);
                float value = valueStr.toFloat();
                buffer = ""; // Reset buffer

                // Bỏ qua giá trị 0.00000
                if (value == 0.0f) {
                    collecting = false;
                    sampleIndex = 0;
                    Serial.println("Received: 0.00000, reset collecting");
                    continue;
                }

                // Bắt đầu thu thập nếu chưa
                if (!collecting) {
                    collecting = true;
                    sampleIndex = 0;
                    Serial.println("Started collecting samples");
                }

                // Lưu mẫu vào mảng
                if (sampleIndex < SAMPLE_COUNT) {
                    samples[sampleIndex++] = value;
                    Serial.print("Sample ");
                    Serial.print(sampleIndex);
                    Serial.print(": ");
                    Serial.println(value);
                }

                // Đã đủ 10 mẫu
                if (sampleIndex == SAMPLE_COUNT) {
                    float mostFrequent = findMostFrequent(samples, SAMPLE_COUNT);

                    // Chuyển float thành chuỗi
                    char result[20];
                    snprintf(result, sizeof(result), "%.5f", mostFrequent);

                    // Gửi vào EventQueue
                    EventQueue_Push(1, result); // 1 là EVENT_LOADCELL_CHANGED giả định

                    // Reset
                    collecting = false;
                    sampleIndex = 0;
                    Serial.println("Collection complete, reset for next cycle");
                }
            } else {
                buffer = ""; // Reset buffer nếu chỉ nhận dấu '='
                Serial.println("Received: <empty>");
            }
        }
    }
}

void setup() {
    Serial.begin(9600);
    SerialPort1.begin(9600, SERIAL_8N1, RS485_RX_1, RS485_TX_1);

    Serial.println("RS232 Initialized");
}

void loop() {

    // Nhận và xử lý dữ liệu
    receiveRS232Data();
}
